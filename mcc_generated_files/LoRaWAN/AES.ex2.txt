#include <p18CXXX.h>
#include "AES_ASM.h"
#include "AESdef.inc"

/****************************** Constants ***************************/

void main(void)
{
	unsigned char i;
	for(i=0;i<16;i++)
	{
		key[i]=i;
		block[i]=i;
	}
	#ifdef ENCODE
	AESEncrypt();
	#endif
	#ifdef DECODE
	AESDecrypt();
	#endif
	Nop();
}
